import React from 'react';
import { Table, Heading } from '@athena/forge';
import faker from 'faker';

const departments = [
  'Avon Main Office',
  'Avon OBGYN',
  'Beaver Creek Cardiology',
  'Cedarville Chronic Care Management',
  'Findlay OBGYN',
  'Greater Columbus Cardiology',
  'Home Road Primary Care',
  'Lima Main Office',
  'Parma Main Office',
  'Vermillion Cardiology',
];

const data = departments.map((department, index) => {
  return {
    id: index,
    department: department,
    hold: faker.finance.amount(0, 1000),
    mgrHold: faker.finance.amount(0, 1000),
    cboHold: faker.finance.amount(0, 1000),
  };
});

const currencyTemplate = (num) =>
  '$' + num.toLocaleString('en-US', { style: 'currency', currency: 'USD' });

export default function TableDemo() {
  return (
    <div className="fe_u_margin--large" >
      <Heading variant="section" text="Hold Summary"/>
      <Table
        layout="compact"
        rows={data}
        columns={[
          {
            key: 'department',
            displayName: 'Department',
          },
          {
            key: 'hold',
            displayName: 'Hold',
            numeric: true,
            template: currencyTemplate,
          },
          {
            key: 'mgrHold',
            displayName: 'Mgr Hold',
            numeric: true,
            template: currencyTemplate,
          },
          {
            key: 'cboHold',
            displayName: 'CBO Hold',
            numeric: true,
            template: currencyTemplate,
          },
        ]}
      />
    </div>
  );
}
