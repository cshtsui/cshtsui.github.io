import React, { Component } from 'react';
import { Form, MultiField } from '@athena/forge';

class MultiFieldDemo extends Component {

  constructor(props) {
    super(props);
    this.state = { formValues: '' };
    this.handleSubmit = this.handleSubmit.bind(this);
  }

  handleSubmit(event){
    event.preventDefault();
    const values = {};
    Array.prototype.forEach.call(event.target.elements, function(el){
      if (el.value) {
        values[el.name] = el.value;
      }
    });
    this.setState({ formValues: JSON.stringify(values, null, 2) });
  }

  render(){
    return (
      <div className="fe_u_padding--large">
        <Form onSubmit={this.handleSubmit}>

          <MultiField
            fields={[
              {
                labelText: 'First name',
                name: 'firstName',
              },
              {
                labelText: 'MI',
                name: 'middleInitial',
                inputWidth: '5ch',
              },
              {
                labelText: 'Last name',
                name: 'lastName',
              },
            ]}
          />

        </Form>

        <div style={{
          boxSizing: 'border-box',
          fontFamily: 'monospace',
          color: 'dimgray',
          border: '1px solid gainsboro',
          background: 'whitesmoke',
          padding: '1rem',
          borderRadius: '3px',
          width: '300px',
          minHeight: '62px',
        }}>
          <strong>FORM DATA:</strong><br />
          <code><pre>{this.state.formValues}</pre></code>
        </div>

      </div>
    );
  }
}

export default MultiFieldDemo;
