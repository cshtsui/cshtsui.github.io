import React, { Component } from 'react';
import { createStore, combineReducers } from 'redux';
import { reducer as formReducer } from 'redux-form';
import { Provider } from 'react-redux';
import moment from 'moment';
import {
  ReduxForm,
  ReduxFormField,
  ReduxMultiField,
  DateInput
} from '@athena/forge';

const rootReducer = combineReducers({
  form: formReducer,
});

const store = createStore(
  rootReducer
);

class MultiFieldDefault extends Component {

  constructor(props) {
    super(props);
    this.state = { formValues: '' };
    this.handleSubmit = this.handleSubmit.bind(this);
  }

  handleSubmit(values){
    this.setState({ formValues: JSON.stringify(values, null, 2) });
  }

  render(){
    return (
      <Provider store={store}>
        <div className="fe_u_padding--large">
          <ReduxForm form="redux-form-field-demo" onSubmit={this.handleSubmit}>
            <ReduxMultiField
              fields={[
                {
                  id: 'firstName',
                  labelText: 'First name',
                  required: true,
                },
                {
                  id: 'middleInitial',
                  labelText: 'MI',
                  inputWidth: '5ch',
                },
                {
                  id: 'lastName',
                  labelText: 'Last name',
                  required: true,
                },
              ]}
              nested={true}
            />
            <ReduxFormField
              id="dob"
              labelText="Date of birth"
              labelAlwaysAbove={true}
              inputAs={DateInput}
              showCalendar={false}
              selectYearStart={moment.utc().year() - 100}
              selectYearEnd={moment.utc().year()}
              constraints={{
                date: {
                  latest: moment.utc().startOf('day'),
                  message: `This field only accepts past dates.`,
                },
              }}
              disabledDays={{
                after: moment.utc(),
              }}
              required
            />
          </ReduxForm>
          <div style={{
            boxSizing: 'border-box',
            fontFamily: 'monospace',
            color: 'dimgray',
            border: '1px solid gainsboro',
            background: 'whitesmoke',
            padding: '1rem',
            borderRadius: '3px',
            width: '300px',
            minHeight: '62px',
          }}>
            <strong>FORM DATA:</strong><br />
            <code><pre>{this.state.formValues}</pre></code>
          </div>

        </div>
      </Provider>
    );
  }

}

export default MultiFieldDefault;
