import React, { Component } from 'react';
import Toast from '@athena/forge-toast/build/components/toast/toast';
import faker from 'faker';
import moment from 'moment';
import omit from 'lodash/omit';

class LabResults extends Component {
  constructor(props) {
    super(props);

    this.state = { results: {} };
  }

  componentDidMount() {
    this.generateLab(); // Add the first lab immediately
    this.generateLabs();
  }

  componentWillUnmount() {
    clearTimeout(this.generationTimeout);
  }

  generateLab() {
    const gender = Math.random() > 0.5;
    const dob = faker.date.past(80);
    const info = {
      name: `${faker.name.firstName(gender ? 0 : 1)} ${faker.name.lastName()}`,
      dob: dob,
      id: faker.random.number(8999) + 1000, // 4 digit number
      gender: gender ? 'M' : 'F',
      age: new Date().getFullYear() - dob.getFullYear(),
    };

    this.setState(prevState => {
      const newResults = prevState;
      newResults[info.id] = info;
      return ({ results: newResults });
    });
  }

  generateLabs() {
    this.generationTimeout = setTimeout(() => {
      this.generateLab();

      this.generateLabs(); // Recursion will be stopped by clearTimeout
    }, 2000 + (Math.random() * 18000)); // 2 - 20 seconds
  }

  handleDismiss(id) {
    this.setState(prevState => ({ results: omit(prevState.results, id) }));
  }

  render() {
    return (
      <div style={{
        background: 'url("/images/forge-toast/schedule-testPractice.png")',
        backgroundPosition: 'bottom left',
        height: '300px',
      }}>
        { Object.entries(this.state.results).map(([ id, info ]) => (
          <Toast
            id={id}
            key={id}
            headerText="Lab Result"
            action={{ label: 'View', onAction: () => {} }}
            alertType="info"
            onDismiss={() => this.handleDismiss(id)}
            onNeverShown={() => this.handleDismiss(id)}
          >
            <strong style={{ display: 'block' }}>{info.name}</strong>
            {`${info.age}yo ${info.gender} | ${moment(info.dob).format('MM-DD-YYYY')} | ${info.id}`}
          </Toast>
        ))}
      </div>
    );
  }
}

export default LabResults;
