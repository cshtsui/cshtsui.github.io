import React, { Component } from 'react';
import { Button, Heading, Stepper } from '@athena/forge';


const STEPS = [
  'Rank and Confirm List',
  'Credentialing Status',
  'Current Remittance Method',
  'Current Payment Method',
  'Review Submission',
];


export default class StepperDemo extends Component {
  constructor(props) {
    super(props);

    this.state = { currentStep: 0 };

    this.onBack = this.onBack.bind(this);
    this.onNext = this.onNext.bind(this);
    this.onSelect = this.onSelect.bind(this);
    this.nextButtonText = this.nextButtonText.bind(this);
    this.getNavigableSteps = this.getNavigableSteps.bind(this);
  }

  onBack () {
    if (this.state.currentStep > 0) {
      this.setState({ currentStep: this.state.currentStep - 1 });
    }
  }

  onNext () {
    if (this.state.currentStep < STEPS.length - 1) {
      this.setState({ currentStep: this.state.currentStep + 1 });
    }
  }

  onSelect (index) {
    this.setState({ currentStep: index });
  }

  nextButtonText (index) {
    if (index === STEPS.length - 1) {
      return 'Submit';
    }
    else {
      return 'Next';
    }
  }

  getNavigableSteps () {
    const steps = [];
    for (let index = 0; index < STEPS.length; index++) {
      steps.push(index < this.state.currentStep);
    }
    return steps;
  }

  render() {
    const steps = STEPS;
    const currentStep = this.state.currentStep;

    return (
      <div>
        <div className="fe_u_padding--medium">
          <Heading text="Payer List" variant="page" />
          <Stepper
            selected={this.state.currentStep}
            onSelect={this.onSelect}
            steps={steps}
            navigableSteps={this.getNavigableSteps()}
            numerical
          />
        </div>
        <div className="fe_u_padding--medium">
          <Heading text={steps[currentStep]} variant="section" />
          <p>Page content for the {steps[currentStep]} step would appear here.</p>
          <Button text="Back" variant="secondary" onClick={this.onBack} disabled={currentStep === 0}/>
          <Button className="fe_u_margin--left-medium" text={this.nextButtonText(currentStep)} onClick={this.onNext} />
        </div>
      </div>
    );
  }
}
