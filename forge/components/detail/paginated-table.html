import React from 'react';
import { Heading, Paginator, Table } from '@athena/forge';
import faker from 'faker';
import moment from 'moment';
import _ from 'lodash';

const pageData = (data, selectedIndex, pageSize) => (
  data.slice(selectedIndex * pageSize, (selectedIndex + 1) * pageSize)
);

const providers = [];
for (let i = 0; i < 10; i++) {
  providers.push(faker.name.lastName());
}

let searchResults = [];
for (let i = 0; i < 68; i++) {
  searchResults.push({
    lastName: 'Smith',
    firstName: faker.name.firstName(),
    dateOfBirth: moment(faker.date.past(100)),
    lastVisit: moment(faker.date.past(1)),
    provider: faker.random.arrayElement(providers),
    id: faker.random.number({ min: 100000, max: 999999 }),
  });
}
searchResults = _.orderBy(searchResults, 'firstName');

class PaginatedTable extends React.Component {
  constructor(props) {
    super(props);
    this.state = { pageIndex: 0 };
    this.pageSize = 5;
    this.updatePage = this.updatePage.bind(this);
  }

  updatePage(index) {
    this.setState({ pageIndex: index });
  }

  render() {
    return (
      <div
        className="fe_u_flex-container fe_u_flex-justify-content--space-between fe_u_margin--large"
        style={{ maxWidth: 1000, flexWrap: 'wrap' }}
      >
        <Heading
          text={'Patient results for "Smith"'}
          variant="section"
          className="fe_u_margin--none"
        />
        <Paginator
          displayMode="withPageJump"
          pageCount={Math.ceil(searchResults.length * 1.0 / this.pageSize)}
          selectedIndex={this.state.pageIndex}
          onSelectedIndexChange={this.updatePage}
        />
        <Table
          className="fe_u_fill--width fe_u_margin--top-small"
          columns={[
            { key: 'lastName', displayName: 'Last Name' },
            { key: 'firstName', displayName: 'First Name' },
            { key: 'id', displayName: 'ID #', numeric: true },
            { key: 'provider', displayName: 'PCP' },
          ]}
          rows={pageData(searchResults, this.state.pageIndex, this.pageSize)}
        />
      </div>
    );
  }
}

export default PaginatedTable;
