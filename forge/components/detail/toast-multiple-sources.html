import React, { Component } from 'react';
import { Heading, Form, FormField, GridCol, GridRow } from '@athena/forge';
import Toast from '@athena/forge-toast/build/components/toast/toast';
import uniqueId from 'lodash/uniqueId';

class ToastSource extends Component {
  constructor(props) {
    super(props);

    this.state = {};
  }

  componentWillUnmount() {
    clearTimeout(this.generationTimeout);
  }

  handleDismiss(id) {
    this.setState(prevState => (prevState.id === id
      ? { firstName: undefined, lastName: undefined, id: undefined }
      : {}
    ));
  }

  render() {
    const isCreate = this.props.type === 'create';
    return (
      <div>
        <Heading
          headingTag="h2"
          variant="section"
          text={`${isCreate ? 'Create' : 'Delete'} a Charge`}
          headingDescription={isCreate ? `but it will fail` : `and it will succeed`}
        />
        <Form
          onSubmit={event => {
            if (event.target[0].value && event.target[1].value) {
              this.setState({
                firstName: event.target[0].value,
                lastName: event.target[1].value,
                id: uniqueId(),
              });
            }
            event.preventDefault();
          }}
          buttonText={isCreate ? 'Create' : 'Delete'}
        >
          <FormField
            labelAlwaysAbove
            labelText="First name"
            required
          />
          <FormField
            labelAlwaysAbove
            labelText="Last name"
            required
          />
        </Form>
        { this.state.id &&
          <Toast
            id={this.state.id}
            alertType={isCreate ? 'attention' : 'success'}
            headerText={isCreate ? 'Charge Not Created' : 'Charge Deleted'}
            action={isCreate
              ? { label: 'View', onAction: () => {} }
              : undefined
            }
          >
            {this.props.type === 'create'
              ? `There was an issue creating the charge for ${this.state.lastName.toUpperCase()}, ${this.state.firstName}.`
              : `You have successfully deleted a charge for ${this.state.lastName.toUpperCase()}, ${this.state.firstName}.`
            }
          </Toast>
        }
      </div>
    );
  }
}

const MultipleSources = (props) => (
  <div className="fe_u_padding--large">
    <GridRow>
      <GridCol width={{ small: 6 }}>
        <ToastSource type="create" />
      </GridCol>
      <GridCol width={{ small: 6 }}>
        <ToastSource type="delete" />
      </GridCol>
    </GridRow>
  </div>
);

export default MultipleSources;
