import React from 'react';
import { Heading, List, ListItem, RemovableTag } from '@athena/forge';
import faker from 'faker';

const departmentList = [ 'Internal Medicine', 'Pediatrics', 'Obstetrics' ];

const providersByDepartment = {};
for (let i = 0; i < 10; i++) {
  const departmentName = departmentList[i % departmentList.length];
  if (!providersByDepartment[departmentName]) {
    providersByDepartment[departmentName] = [];
  }
  providersByDepartment[departmentName].push(`${faker.name.firstName()} ${faker.name.lastName()}, MD`);
}

class RemovableTagDemo extends React.Component {
  constructor(props) {
    super(props);
    this.state = { viewingDepartments: departmentList };
    this.onRemove = this.onRemove.bind(this);
    this.resetDemo = this.resetDemo.bind(this);
  }

  resetDemo() {
    this.setState({ viewingDepartments: departmentList });
  }

  onRemove(tagName) {
    return () => {
      const viewingDepartments = this.state.viewingDepartments.slice(0);
      viewingDepartments.splice(viewingDepartments.indexOf(tagName), 1);
      this.setState({ viewingDepartments });
    };
  }

  render() {
    let providers = [];
    for (let i = 0; i < this.state.viewingDepartments.length; i++) {
      providers = providers.concat(providersByDepartment[this.state.viewingDepartments[i]]);
    }
    providers.sort();

    return (
      <div className="fe_u_flex-container">
        <div className="fe_u_padding--large" style={ { minWidth: '200px' } }>
          <Heading id="filter-department" className="fe_u_font-size--default fe_u_font-weight--regular fe_u_color--muted fe_u_margin--bottom-small" text="Filter" />
          <div className="fe_u_font-size--large fe_u_font-weight--semibold fe_u_padding--bottom-medium">Specialty</div>
          <ul className="fe_u_list-style--no-bullet" role="list" aria-labelledby="filter-department">
            { this.state.viewingDepartments.map((name, index) => (
              <li className="fe_u_padding--top-xsmall fe_u_padding--bottom-xsmall" key={`department-${index}`} role="listitem">
                <RemovableTag text={name} onRemove={this.onRemove(name)} />
              </li>
            ))}
          </ul>
          <a onClick={this.resetDemo}>Reset</a>
        </div>
        <div className="fe_u_flex--grow fe_u_padding--large" style={ { minHeight: '380px', borderLeft: '1px solid #D9D9D9' } }>
          <Heading text="Providers" />
          <List padded={false}>
            { providers.map((name, index) => (
              <ListItem className="fe_u_flex-container fe_u_flex-align-items--center" key={`provider-${index}`}>{name}</ListItem>
            ))}
          </List>
        </div>
      </div>
    );
  }
}

export default RemovableTagDemo;
