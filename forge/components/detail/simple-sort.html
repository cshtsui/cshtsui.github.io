import { Table } from '@athena/forge';

const columns = [

    key: 'col2',

    col1: 'ABCDEF',

  },
];

export default function TableSimpleSortDemo() {
  return (
    <div className="fe_u_margin--large" >
      <Table

        rows={rows}
        columns={columns}
      />
    </div>
  );
}
