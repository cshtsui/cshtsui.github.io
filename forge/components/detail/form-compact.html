import React, { Component } from 'react';
import moment from 'moment';

import Checkbox from '@athena/forge/lib/checkbox';
import DateInput from '@athena/forge/lib/date-input';
import Form from '@athena/forge/lib/form';
import FormField from '@athena/forge/lib/form-field';
import GridCol from '@athena/forge/lib/grid-col';
import GridRow from '@athena/forge/lib/grid-row';
import Heading from '@athena/forge/lib/heading';
import RadioGroup from '@athena/forge/lib/radio-group';
import Textarea from '@athena/forge/lib/textarea';
import TextInput from '@athena/forge/lib/input';

const FormFields = (props) => (
  <React.Fragment>
    <FormField
      id={`name-${props.variant}`}
      labelText="Name"
    />
    <FormField
      id={`dob-${props.variant}`}
      labelText="Date of birth"
      inputAs={DateInput}
      showCalendar={false}
      selectYearStart={moment.utc().year() - 100}
      selectYearEnd={moment.utc().year()}
    />
    <FormField
      id={`gender-${props.variant}`}
      inputAs={RadioGroup}
      labelText="Gender"
      options={[ 'Male', 'Female', 'Other' ]}
    />
    <FormField
      id={`email-${props.variant}`}
      inputAs={TextInput}
      labelText="Email"
      constraints={ { email: { message: 'Please enter a valid email' } } }
    />
    <FormField
      id={`notes-${props.variant}`}
      inputAs={Textarea}
      labelText="Notes"
    />
    <FormField
      id={`terms-${props.variant}`}
      inputAs={Checkbox}
      labelText="Terms"
      description="I agree to terms"
      required
    />
  </React.Fragment>
);

class FormCompact extends Component {
  render(){
    return (
      <div className="fe_u_padding--large">
        <GridRow>
          <GridCol width={{ small: 6 }} style={{ borderRight: '1px solid #B3B3B3' }}>
            <Heading headingTag="h2" variant="section" text="Regular Form" />
            <Form>
              <FormFields variant="regular" />
            </Form>
          </GridCol>
          <GridCol width={{ small: 6 }} >
            <Heading headingTag="h2" variant="section" text="Compact Form" />
            <Form compact>
              <FormFields variant="compact" />
            </Form>
          </GridCol>
        </GridRow>
      </div>
    );
  }
}

export default FormCompact;
