import React from 'react';
import {
  Card,
  Heading
} from '@athena/forge';

const cardLinkStyles = {
  width: '250px',
};

const CardMedia = (props) => (
  <div style={{
    backgroundImage: `url(${props.image})`,
    height: '250px',
  }}>
    <div
      className="
        fe_u_fill--parent
        fe_u_padding--top-xlarge
        fe_u_padding--bottom-xlarge
        fe_u_padding--left-medium
        fe_u_padding--right-medium
        fe_u_color--invert
        fe_u_font-size--xlarge
      "
      style={{
        background: 'rgba(0, 0, 0, 0.5)',
      }}>{props.text}</div>
  </div>
);

const CardInfo = (props) => (
  <div className="fe_u_padding--small fe_u_flex-container">
    <div
      className="fe_u_margin--right-small"
      style={{
        height: '50px',
        width: '50px',
        borderRadius: '25px',
        overflow: 'hidden',
      }}
    >
      <img src={props.logo} className="fe_u_fill--width" />
    </div>
    <div>
      <Heading variant="subsection" text={props.heading} className="fe_u_margin--bottom-none" />
      <span className="fe_u_color--light fe_u_font-weight--light">May 2, 2017</span>
    </div>
  </div>
);

// image paths
const cdcBg = '../../assets/images/card-demo/cdc_card_bg.png';
const cdcLogo = '../../assets/images/card-demo/cdc_logo.png';
const mayoBg = '../../assets/images/card-demo/mayo_card_bg.png';
const mayoLogo = '../../assets/images/card-demo/mayo_logo.png';
const nihBg = '../../assets/images/card-demo/nih_card_bg.png';
const nihLogo = '../../assets/images/card-demo/nih_logo.png';

export default function CardBasic () {
  return (
    <div className="fe_u_padding--large" style={{ display: 'flex' }}>
      <Card
        mediaSlot={<CardMedia image={cdcBg} text="Putting the Brakes on Diabetes Complications" />}
        padded={false}
        style={cardLinkStyles}
        className="fe_u_margin--right-large"
      >
        <CardInfo logo={cdcLogo} heading="CDC" date="May 2, 2017" />
      </Card>
      <Card
        mediaSlot={<CardMedia image={mayoBg} text="Definition of Type II Diabetes" />}
        padded={false}
        style={cardLinkStyles}
        className="fe_u_margin--right-large"
      >
        <CardInfo logo={mayoLogo} heading="Mayo Clinic" date="July 1, 2018" />
      </Card>
      <Card
        mediaSlot={<CardMedia image={nihBg} text="IVIG: The Basics" />}
        padded={false}
        style={cardLinkStyles}
      >
        <CardInfo logo={nihLogo} heading="NIH" date="May 2, 2017" />
      </Card>
    </div>
  );
}
