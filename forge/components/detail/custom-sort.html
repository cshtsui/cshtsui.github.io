import React from 'react';
import { Table } from '@athena/forge';

const columns = [
  {
    key: 'name',
    displayName: 'Name',
    sortable: { reducer: name => name.last },
    template: name => name.first + ' ' + name.last,
  },
  {
    key: 'height',
    displayName: 'Height',
    numeric: true,
    sortable: { reducer: height => height.feet * 12 + height.inches },
    template: height => height.feet + '\' ' + height.inches + '"',
  },
];

const rows = [
  {
    id: 1,
    name: { first: 'Richard', last: 'Grayson' },
    height: { feet: 5, inches: 4 },
  },
  {
    id: 2,
    name: { first: 'Rachel', last: 'Roth' },
    height: { feet: 5, inches: 4 },
  },
  {
    id: 3,
    name: { first: 'Victor', last: 'Stone' },
    height: { feet: 6, inches: 0 },
  },
  {
    id: 4,
    name: { first: 'Garfield', last: 'Logan' },
    height: { feet: 5, inches: 0 },
  },
  {
    id: 5,
    name: { first: 'Koriand\'r', last: '' },
    height: { feet: 5, inches: 6 },
  },
];

export default function TableCustomSortDemo() {
  return (
    <div className="fe_u_margin--large" >
      <Table
        layout="default"
        rows={rows}
        columns={columns}
      />
    </div>
  );
}
