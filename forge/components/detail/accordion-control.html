import React from 'react';
import {
  Accordion,
  AccordionItem,
  Heading,
  Form,
  FormField,
  RadioGroup,
  Button
} from '@athena/forge';

const FrequencyInput = (props) => (
  <RadioGroup {...props} options={[
    { text: 'Daily', value: 2 },
    { text: 'Less than daily', value: 1 },
    { text: 'Not at all', value: 0 },
    { text: 'Don\'t know', value: -1 },
  ]}/>
);

const questions = [
  {
    id: 'current-smoke',
    heading: 'Current tobacco smoking status',
    label: 'Do you currently smoke tobacco on a daily basis,'
      + ' less than daily, or not at all?',
  },
  {
    id: 'past-smoke',
    heading: 'Past tobacco smoking status',
    label: 'In the past, have you smoked tobacco on a daily basis,'
    + ' less than daily, or not at all?',
  },
  {
    id: 'current-non-smoke',
    heading: 'Current smokeless tobacco use',
    label: 'Do you currently use smokeless tobacco on a daily basis,'
    + ' less than daily, or not at all?',
  },
  {
    id: 'past-non-smoke',
    heading: 'Past smokeless tobacco use',
    label: 'In the past, have you used smokeless tobacco on a daily basis,'
    + ' less than daily, or not at all?',
  },
];

export default class AccordionControl extends React.Component {
  constructor(props) {
    super(props);

    this.state = { expandedIndex: 0 };
  }

  render() {
    return (
      <div className="fe_u_padding--large">
        <Heading
          text="Tobacco Use Survey"
          variant="section"
          headingTag="h2"
          className="fe_u_padding--left-large"
          subheading="Please respond in each section before submitting"
        />
        <Form includeSubmitButton={false} onSubmit={(event) => event.preventDefault()} >
          <Accordion>
            { questions.map((question, index) => (
              <AccordionItem
                expanded={this.state.expandedIndex === index}
                onExpandedChange={() => this.setState({ expandedIndex: index })}
                key={question.id}
                headingText={question.heading}
                mountedWhileHidden>
                <FormField
                  labelText={question.label}
                  labelWidth={12}
                  inputAs={FrequencyInput}
                  id={question.id}
                />
                {
                  index === (questions.length - 1)
                    ? <Button text="Submit" type="submit" />
                    : (
                      <Button text="Next"
                        onClick={() => this.setState({ expandedIndex: index + 1 })}
                      />
                    )
                }
              </AccordionItem>
            ))}
          </Accordion>
        </Form>
      </div>
    );
  }
}
