import React from 'react';
import { createStore, combineReducers } from 'redux';
import { reducer as formReducer } from 'redux-form';
import { Provider } from 'react-redux';
import moment from 'moment';
import {
  DateInput,
  ReduxForm,
  ReduxFormField,
  Textarea
} from '@athena/forge';

const store = createStore(
  combineReducers({
    form: formReducer,
  })
);

class FormValidation extends React.Component {
  constructor(props) {
    super(props);
    this.handleSubmit = this.handleSubmit.bind(this);
    this.state = { dateHintText: undefined };
  }

  handleSubmit(values) {
    // do something with form submission here
  }

  render() {
    return (
      <Provider store={store}>
        <ReduxForm
          form="validation-demo"
          onSubmit={this.handleSubmit}
          className="fe_u_padding--large"
        >
          <ReduxFormField
            labelText="Required field"
            hintText="Marking a field as required is a basic form of validation."
            required
          />

          <ReduxFormField
            labelText="Email"
            constraints={{
              email: { message: `Email address should be in "name@company.com" format.` },
            }}
          />

          <ReduxFormField
            labelText="Date"
            inputAs={DateInput}
            constraints={{
              date: {
                earliest: moment.utc().startOf('day'),
                message: `This field only accepts future dates.`,
              },
            }}
            onHintTextChange={
              (hintText) => this.setState({ dateHintText: hintText })
            }
            hintText={ this.state.dateHintText || `Future dates only` }
            disabledDays={{
              before: moment.utc(),
            }}
            selectYearStart={moment.utc().year()}
          />

          <ReduxFormField
            labelText="ZIP code (format)"
            constraints={{
              format:
                {
                  pattern: /\d{5}(-\d{4})?/,
                  message: `ZIP code should be 5 digits (e.g., 12345) or 9 digits with a dash (e.g, 12345-6789).`,
                },
            }}
          />

          <ReduxFormField
            labelText="Length limit"
            inputAs={Textarea}
            constraints={{ length: {
              maximum: 140,
              message: `This field is limited to 140 characters.`,
            } }}
            hintText="Twitter may allow more than 140 characters, but we don't."
          />

          <ReduxFormField
            labelText="Numericality"
            constraints={{ numericality: {
              onlyInteger: true,
              message: `This field only accepts integers.`,
            } }}
          />

          <ReduxFormField
            labelText="Custom"
            constraints={{
              letterE: (value) => value && /e/i.test(value) ? `This field doesn't allow the letter E.` : undefined,
            }}
            hintText="Custom constraint prohibiting the letter E."
          />
        </ReduxForm>
      </Provider>
    );
  }

}

export default FormValidation;
