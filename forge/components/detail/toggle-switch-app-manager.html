import React, { Component } from 'react';
import { GridCol, GridRow, Heading } from '@athena/forge';
import ToggleSwitch from '@athena/forge-toggle-switch/build/components/toggle-switch/toggle-switch';

class AppManager extends Component {
  constructor(props) {
    super(props);
    this.renderAppRow = this.renderAppRow.bind(this);
  }

  renderAppRow(apps, descriptions) {
    return apps.map((app, index) => (
      <GridRow key={index}>
        <GridCol width={{ small: 6, medium: 6, large: 6 }} >
          <span className="toggleswitch-apps-demo__app">{app.name}</span>
        </GridCol>
        <GridCol width={{ small: 6, medium: 6, large: 6 }} >
          <ToggleSwitch defaultChecked={app.defaultChecked} descriptions={descriptions}/>
        </GridCol>
      </GridRow>
    ));
  }

  render() {
    const descriptions = {
      checked: 'Enabled',
      unchecked: 'Disabled',
    };

    const apps = [
      {
        name: 'Epocrates',
        defaultChecked: true,
      },
      {
        name: 'Praxify',
        defaultChecked: true,
      },
      {
        name: 'athenaText',
        defaultChecked: false,
      },
    ];

    return (
      <div className='fe_u_padding--large'>
        <Heading variant='section' text='Apps Management' />
        { this.renderAppRow(apps, descriptions) }
      </div>
    );
  }
}

export default AppManager;
