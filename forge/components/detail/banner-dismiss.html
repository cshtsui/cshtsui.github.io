import React, { Component } from 'react';
import { Banner, BannerItem, Button } from '@athena/forge';
import { CSSTransition } from 'react-transition-group';

class BannerDismiss extends Component {
  constructor(props) {
    super(props);

    this.state = { showBanner: true };

    this.toggleBanner = this.toggleBanner.bind(this);
  }

  toggleBanner () {
    this.setState({ showBanner: !this.state.showBanner });
  }

  // the height of an element cannot be animated
  // unless explicity set. Transition lifecycle
  // callbacks are used to set inline styles for height.

  handleEnter(elem){
    elem.style.height = '0';
  }

  handleEntering(elem){
    const height = elem.scrollHeight + 'px';
    elem.style.height = height;
  }

  handleEntered(elem){
    elem.style.height = null;
  }

  handleExit(elem){
    const height = elem.scrollHeight + 'px';
    elem.style.height = height;
  }

  handleExiting(elem){
    elem.style.height = '0';
  }

  render() {
    return (
      <div className="fe_u_padding--large">
        {/* the css transition relies on the following external classes
            that conform to react-transition-group's naming convention
            https://reactcommunity.org/react-transition-group/css-transition#CSSTransition-prop-classNames

          .fade-open-appear,
          .fade-open-enter {
            opacity: 0;
            overflow: hidden;
          }

          .fade-open-appear-active,
          .fade-open-enter-active {
            opacity: 1;
            transition:
              opacity 400ms cubic-bezier(0.215, 0.61, 0.355, 1),
              height 400ms cubic-bezier(0.215, 0.61, 0.355, 1);
          }

          .fade-open-exit {
            opacity: 1;
            overflow: hidden;
          }

          .fade-open-exit-active {
            opacity: 0;
            transition:
              opacity 400ms cubic-bezier(0.215, 0.61, 0.355, 1),
              height 400ms cubic-bezier(0.215, 0.61, 0.355, 1);
          }
        */}
        <CSSTransition
          in={this.state.showBanner}
          appear={true}
          timeout={400}
          classNames="fade-open"
          onEnter={this.handleEnter}
          onEntering={this.handleEntering}
          onEntered={this.handleEntered}
          onExit={this.handleExit}
          onExiting={this.handleExiting}
          unmountOnExit={true}
        >
          {state => (
            <Banner onHide={this.toggleBanner} alertType="attention">
              <BannerItem headerText="Unable to Process">
                There was a problem processing the uploaded report. Please try again.
              </BannerItem>
            </Banner>
          )}
        </CSSTransition>
        <div className="fe_u_padding--top-large">
          <Button text="Toggle Banner" onClick={this.toggleBanner} />
        </div>
      </div>
    );
  }
}

export default BannerDismiss;
