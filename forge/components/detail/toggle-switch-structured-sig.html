import React, { Component } from 'react';
import {
  FormField,
  Heading,
  MultiField,
  ReadOnlyInput,
  Select
} from '@athena/forge';
import ToggleSwitch from '@athena/forge-toggle-switch/build/components/toggle-switch/toggle-switch';


class StructuredSig extends Component {
  constructor(props) {
    super(props);
    this.state = {
      showStructuredSig: true,
    };
    this.onSigToggleChange = this.onSigToggleChange.bind(this);
  }

  renderStructuredSig() {
    return (
      <FormField
        labelText='Sig'
        inputAs={MultiField}
        fields={[ {
          inputAs: ReadOnlyInput,
          text: 'Take',
          labelText: '',
        }, {
          inputWidth: '4ch',
          placeholder: '1',
        }, {
          inputAs: Select,
          defaultValue: 'tablet(s)',
          labelText: '',
          options: [ 'tablet(s)' ],
        }, {
          inputAs: Select,
          defaultValue: 'everyday',
          labelText: '',
          options: [ 'everyday' ],
        }, {
          inputAs: ReadOnlyInput,
          text: 'by oral route',
          labelText: '',
        }, {
          inputAs: Select,
          defaultValue: 'with meals',
          labelText: '',
          options: [ 'with meals' ],
        }, {
          inputAs: ReadOnlyInput,
          text: 'for',
          labelText: '',
        }, {
          inputWidth: '5ch',
          placeholder: '30',
        }, {
          inputAs: ReadOnlyInput,
          text: 'days',
          labelText: '',
        } ]}
      />
    );
  }

  renderUnstructuredSig() {
    return (
      <FormField
        labelText='Sig'
        placeholder='1'
      />
    );
  }

  onSigToggleChange(e) {
    this.setState({
      showStructuredSig: e.target.value,
    });
  }

  render() {
    return (
      <div className='fe_u_padding--large'>
        <Heading variant='subsection' text='aspirin 81mg tablet, delayed release' />
        <div>
          <FormField
            labelText='Structured sig'
            inputAs={ToggleSwitch}
            onChange={this.onSigToggleChange}
            checked={this.state.showStructuredSig}
          />
          { this.state.showStructuredSig ? this.renderStructuredSig() : this.renderUnstructuredSig() }
        </div>
      </div>
    );
  }
}

export default StructuredSig;
