import React from 'react';
import { DateInput, Heading, Paginator, List, ListItem } from '@athena/forge';
import faker from 'faker';
import moment from 'moment';
import _ from 'lodash';

const pageData = (data, selectedIndex, pageSize) => (
  data.slice(selectedIndex * pageSize, (selectedIndex + 1) * pageSize)
);

const providers = [];
for (let i = 0; i < 10; i++) {
  providers.push(faker.name.lastName());
}

let searchResults = [];

for (let i = 0; i < 137; i++) {
  searchResults.push({
    lastName: 'Smith',
    firstName: faker.name.firstName(),
    dateOfBirth: moment(faker.date.past(100)),
    lastVisit: moment(faker.date.past(1)),
    provider: faker.random.arrayElement(providers),
    id: faker.random.number({ min: 100000, max: 999999 }),
  });
}

searchResults = _.orderBy(searchResults, 'firstName');

class PaginatedList extends React.Component {
  constructor(props) {
    super(props);
    this.state = { pageIndex: 0 };
    this.pageSize = 20;
    this.updatePage = this.updatePage.bind(this);
  }

  updatePage(index) {
    this.setState({ pageIndex: index });
  }

  render() {
    return (
      <div
        className="fe_u_flex-container fe_u_flex-align-items--center fe_u_margin--large"
        style={{ flexDirection: 'column', width: 240, maxHeight: '90vh', border: '1px solid black', overflow: 'auto' }}
      >
        <Paginator
          displayMode="compact"
          pageCount={Math.ceil(searchResults.length * 1.0 / this.pageSize)}
          selectedIndex={this.state.pageIndex}
          onSelectedIndexChange={this.updatePage}
          className="fe_u_margin--xsmall"
          style={{ flex: 'none' }}
        />
        <List className="fe_u_fill--width" dividers >
          {
            pageData(searchResults, this.state.pageIndex, this.pageSize).map(result => (
              <ListItem key={result.id}>
                <Heading text={result.lastName + ', ' + result.firstName} variant="subsection" className="fe_u_margin--bottom-xsmall" />
                <div className="fe_u_flex-container fe_u_flex-justify-content--space-between" >
                  <span>#{result.id}</span>
                  <span><em>DOB:</em> {DateInput.WITHOUT_CONTEXT.formatDate(result.dateOfBirth)}</span>
                </div>
              </ListItem>
            ))
          }
        </List>

        <Paginator
          displayMode="compact"
          pageCount={Math.ceil(searchResults.length * 1.0 / this.pageSize)}
          selectedIndex={this.state.pageIndex}
          onSelectedIndexChange={this.updatePage}
          className="fe_u_margin--xsmall"
          style={{ flex: 'none' }}
        />
      </div>
    );
  }
}

export default PaginatedList;
